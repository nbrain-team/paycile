Rate Limiting

API endpoints are rate limited to ensure fair usage:

  • Authentication endpoints: 5 requests per minute
  • Read operations: 100 requests per minute
  • Write operations: 30 requests per minute
  • AI endpoints: 10 requests per minute

Database Schema

PostgreSQL database structure and relationships

Core Tables

-- Users Table
CREATE TABLE users (
    id UUID DEFAULT uuid_generate_v4() PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    first_name VARCHAR(100) NOT NULL,
    last_name VARCHAR(100) NOT NULL,
    role user_role NOT NULL DEFAULT 'client',
    phone VARCHAR(20),
    company_name VARCHAR(255),
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Cash Flow Transactions Table
CREATE TABLE cash_flow_transactions (
    id UUID DEFAULT uuid_generate_v4() PRIMARY KEY,
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    category_id UUID REFERENCES cash_flow_categories(id),
    date DATE NOT NULL,
    type transaction_type NOT NULL,
    amount DECIMAL(15, 2) NOT NULL,
    description VARCHAR(500) NOT NULL,
    is_recurring BOOLEAN DEFAULT false,
    recurring_frequency recurring_frequency,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Policies Table
CREATE TABLE policies (
    id UUID DEFAULT uuid_generate_v4() PRIMARY KEY,
    policy_number VARCHAR(50) UNIQUE NOT NULL,
    client_id UUID REFERENCES users(id),
    agent_id UUID REFERENCES users(id),
    insurance_company_id UUID REFERENCES insurance_companies(id),
    policy_type VARCHAR(100) NOT NULL,
    premium_amount DECIMAL(12, 2) NOT NULL,
    payment_frequency payment_frequency NOT NULL,
    effective_date DATE NOT NULL,
    expiration_date DATE NOT NULL,
    status VARCHAR(50) NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
